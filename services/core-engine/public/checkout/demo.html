// 1. Load ZenWallet once (CDN / local)
<script src="https://zenpay-jshp.onrender.com/zenwallet-sdk.js"></script>

// 2. Your backend creates the order, returns order_id
// 3. Frontend opens checkout with that order_id
const zen = new ZenWallet({
  key: "pk_test_acme_xxx",
  onSuccess: (res) => {
    // POST res to your backend → verify HMAC → mark paid
    fetch("/verify-payment", { method: "POST", body: JSON.stringify(res) });
  },
  onFailure: (err) => console.error("Payment failed", err)
});

zen.open({ order_id: "order_9b2a1c32e4f" });
